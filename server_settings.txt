/etc/nginx/sites-available/geekshop


server {
    listen 80;
    server_name 89.108.77.224;

    location = /favicon.ico { access_log off; log_not_found off; }
    location /static/ {
            root /home/django/7-Django_Base_geekshop;
    }
    location /media/ {
            root /home/django/7-Django_Base_geekshop;
    }

    location / {
            include proxy_params;
            proxy_pass http://unix:/home/django/7-Django_Base_geekshop/geekshop.sock;
    }
 }




/etc/systemd/system/gunicorn.service


[Unit]
Description=gunicorn daemon
After=network.target

[Service]
User=django
Group=www-data
WorkingDirectory=/home/django/7-Django_Base_geekshop
ExecStart=/home/django/env/bin/gunicorn --access-logfile - --workers 3 --bind unix:/home/django/7-Django_Base_geekshop/geekshop.sock geekshop.wsgi

[Install]
